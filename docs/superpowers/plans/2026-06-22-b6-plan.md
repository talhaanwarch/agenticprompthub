# Backend Part 6 — Teams + RBAC API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add team invites, role enforcement (owner/admin/editor/viewer), member management, and team-scoped API keys so a second user can join a team and be given granular write access.

**Architecture:** All new domain files live under `apps/api/src/teams/` with two sub-domains — `members/` and `invites/`. A new `requireRole` middleware factory lives in `src/shared/middleware/` and is applied to every mutating route across B1–B5. The `requireApiKey` middleware is extended to handle team-scoped keys (`user_id = NULL`). Three new schema additions are generated as a single new Prisma migration: the `team_invites` table, the `admin` role extension on `team_member_roles`, and a `scope` column on `api_keys`.

**Tech Stack:** Express + TypeScript, Prisma ORM (`npx prisma migrate dev` for migrations), Zod for validation, `crypto.randomBytes` for token generation, supertest + real Postgres for integration tests, Jest.

## Global Constraints

- TypeScript strict mode everywhere; no `any` without a comment explaining why
- All source files are `.ts`; no `.js` in `src/`
- All table definitions stay in `prisma/schema.prisma` — single source of truth
- All DB access through repository classes only; services never import `prisma`
- `npx prisma migrate dev --name <descriptive-name>` then `npx prisma migrate deploy` — never hand-edit migration files
- All routes prefixed `/api/v1`; routers mounted in `app.ts` only
- Every exported symbol gets a JSDoc block (summary + `@param` + `@returns` + `@throws`)
- Tests use real Postgres (`TEST_DATABASE_URL`), supertest, no mocks
- No commits without user permission

- [ ] **Step 8: Implement requireRole middleware**

Create `apps/api/src/shared/middleware/require-role.middleware.ts`:

```typescript
import { RequestHandler } from 'express';
import { prisma } from '../db/client';

/**
 * Valid role names for RBAC checks. Must stay in sync with Role enum in schema.prisma.
 */
export type TeamRole = 'owner' | 'admin' | 'editor' | 'viewer';

/**
 * Express middleware factory that checks whether the authenticated user holds
 * at least one of the specified roles in the team identified by req.params.id
 * or req.teamId (params.id takes precedence for team-scoped routes).
 *
 * Must be placed after requireAuth or requireApiKey in the middleware chain.
 *
 * @param allowedRoles - One or more roles; the user must hold at least one.
 * @returns Express RequestHandler that calls next() on success or sends 403.
 * @throws Never — all errors are written to the response directly.
 */
export function requireRole(...allowedRoles: TeamRole[]): RequestHandler {
  return async (req, res, next) => {
    // Team ID comes from route param for team-management routes,
    // otherwise from req.teamId set by auth middleware.
    const teamId = (req.params.id as string | undefined) ?? req.teamId;

    if (!teamId) {
      res.status(403).json({ error: { code: 'FORBIDDEN', message: 'No team context.' } });
      return;
    }

    // Team-key requests have req.userId === null; they cannot pass role checks
    // for member management endpoints. Callers that want to explicitly block
    // team keys should check req.userId === null before calling requireRole.
    const userId = req.user?.id ?? null;

    if (!userId) {
      res.status(403).json({
        error: { code: 'TEAM_KEY_NOT_PERMITTED', message: 'Team API keys cannot manage members.' },
      });
      return;
    }

    const rows = await prisma.teamMemberRole.findMany({
      where: {
        teamMember: {
          userId,
          teamId,
        },
      },
      select: { role: true },
    });

    const userRoles = rows.map((r) => r.role as TeamRole);
    const permitted = allowedRoles.some((r) => userRoles.includes(r));

    if (!permitted) {
      res.status(403).json({ error: { code: 'FORBIDDEN', message: 'Insufficient role.' } });
      return;
    }

    // Attach roles so downstream handlers can inspect without re-querying
    req.userRoles = userRoles;
    next();
  };
}
```

- [ ] **Step 13: Update requireApiKey to handle team-scoped keys**

Open the file that contains `requireApiKey` (from B1 — typically `apps/api/src/shared/middleware/require-auth.middleware.ts`). Replace the inner query and assignment logic:

```typescript
/**
 * Authenticates a request via Bearer API key.
 * Supports both personal keys (scope='personal', user_id set) and
 * team-scoped keys (scope='team', user_id NULL).
 * Sets req.user (null for team keys) and req.teamId.
 *
 * @throws Never — errors are written directly to the response.
 */
export const requireApiKey: RequestHandler = async (req, res, next) => {
  const authHeader = req.headers['authorization'];
  if (!authHeader?.startsWith('Bearer ')) {
    res.status(401).json({ error: { code: 'UNAUTHORIZED', message: 'API key required.' } });
    return;
  }
  const keyValue = authHeader.slice(7);

  const row = await prisma.apiKey.findFirst({
    where: { key: keyValue, revokedAt: null },
    include: { user: { select: { email: true } } },
  });

  if (!row) {
    res.status(401).json({ error: { code: 'UNAUTHORIZED', message: 'Invalid or revoked API key.' } });
    return;
  }

  if (row.scope === 'personal') {
    req.user = { id: row.userId!, email: row.user!.email };
  } else {
    // Team-scoped key: no associated user. req.user remains undefined.
    // requireRole will detect this and return TEAM_KEY_NOT_PERMITTED.
    req.user = undefined;
  }
  req.teamId = row.teamId;
  next();
};
```

import { prisma } from '../../shared/db/client';
import type { TeamInvite, InviteListItem } from './invites.types';

/**
 * All database access for the team_invites table.
 * Services must not import prisma directly — only this repository does.
 */
export class InvitesRepository {
  /**
   * Inserts a new invite row and returns the created record.
   *
   * @param data - The invite data to insert.
   * @returns The newly created TeamInvite row.
   */
  async create(data: {
    teamId: string;
    token: string;
    invitedBy: string;
    roles: string[];
    expiresAt: Date;
  }): Promise<TeamInvite> {
    return prisma.teamInvite.create({ data });
  }

  /**
   * Looks up an invite by its token string.
   *
   * @param token - The 64-char hex token from the invite URL.
   * @returns The invite row, or undefined if not found.
   */
  async findByToken(token: string): Promise<TeamInvite | undefined> {
    return (await prisma.teamInvite.findUnique({ where: { token } })) ?? undefined;
  }

  /**
   * Fetches all pending (unused, not-expired) invites for a team,
   * joined with the inviter's email. Token is excluded from this projection.
   *
   * @param teamId - The team whose pending invites to list.
   * @returns Array of invite list items (token field is not present).
   */
  async findPendingByTeam(teamId: string): Promise<InviteListItem[]> {
    const now = new Date();
    const rows = await prisma.teamInvite.findMany({
      where: {
        teamId,
        usedAt: null,
        expiresAt: { gt: now },
      },
      include: {
        inviter: { select: { id: true, email: true } },
      },
    });

    return rows.map((r) => ({
      id:        r.id,
      roles:     r.roles,
      expiresAt: r.expiresAt.toISOString(),
      createdAt: r.createdAt.toISOString(),
      invitedBy: { id: r.inviter.id, email: r.inviter.email },
    }));
  }

  /**
   * Marks an invite as used by setting used_at to the current time.
   *
   * @param inviteId - The UUID of the invite to consume.
   */
  async markUsed(inviteId: string): Promise<void> {
    await prisma.teamInvite.update({
      where: { id: inviteId },
      data: { usedAt: new Date() },
    });
  }

  /**
   * Hard-deletes a pending invite. Used by the revoke endpoint.
   * Returns false if no row was deleted (not found or already used).
   *
   * @param inviteId - The UUID of the invite.
   * @param teamId   - Must match to prevent cross-team deletion.
   * @returns true if deleted, false if not found.
   */
  async deleteById(inviteId: string, teamId: string): Promise<boolean> {
    const existing = await prisma.teamInvite.findFirst({
      where: { id: inviteId, teamId, usedAt: null },
    });
    if (!existing) return false;
    await prisma.teamInvite.delete({ where: { id: inviteId } });
    return true;
  }
}

// apps/api/src/teams/invites/invites.service.ts
import crypto from 'crypto';
import { InvitesRepository } from './invites.repository';
import { MembersRepository } from '../members/members.repository';
import { AuditRepository } from '../../shared/audit/audit.repository'; // from B5
import type {
  CreateInviteDto,
  InviteCreatedResponse,
  InviteListItem,
  AcceptInviteResponse,
} from './invites.types';
import {
  NotFoundError,
  ConflictError,
  GoneError,
} from '../../shared/errors/http-errors';

const INVITE_TTL_MS = 7 * 24 * 60 * 60 * 1000;

/**
 * Business logic for invite generation, acceptance, listing, and revocation.
 * Coordinates InvitesRepository and MembersRepository; never touches prisma directly.
 */
export class InvitesService {
  constructor(
    private readonly invitesRepo: InvitesRepository,
    private readonly membersRepo: MembersRepository,
    private readonly auditRepo: AuditRepository,
  ) {}

  /**
   * Generates a single-use invite token for the given team with specified roles.
   * Expiry is fixed at 7 days from creation.
   *
   * @param teamId    - Team to invite into.
   * @param actorId   - User ID of the inviter (for audit log).
   * @param dto       - Validated request body including roles array.
   * @returns Created invite with token and expiry.
   */
  async generateInvite(
    teamId: string,
    actorId: string,
    dto: CreateInviteDto,
  ): Promise<InviteCreatedResponse> {
    const token = crypto.randomBytes(32).toString('hex');
    const expiresAt = new Date(Date.now() + INVITE_TTL_MS);

    const invite = await this.invitesRepo.create({
      teamId,
      token,
      invitedBy: actorId,
      roles: dto.roles,
      expiresAt,
    });

    await this.auditRepo.record({
      teamId,
      actorId,
      event: 'member_invited',
      entityType: 'team',
      entityId: teamId,
      metadata: { roles: dto.roles },
    });

    return {
      id:        invite.id,
      token:     invite.token,
      inviteUrl: `/invites/${invite.token}`,
      roles:     invite.roles,
      expiresAt: invite.expiresAt.toISOString(),
    };
  }

  /**
   * Accepts an invite: validates the token, checks for expiry and duplicate
   * membership, then atomically creates the team_member + team_member_roles rows
   * and marks the invite as used.
   *
   * @param token     - Raw token string from the URL path.
   * @param userId    - Authenticated user accepting the invite.
   * @returns Team and roles of the newly added member.
   * @throws {NotFoundError}  If the token does not exist.
   * @throws {ConflictError}  If already used or user already a member.
   * @throws {GoneError}      If the invite has expired.
   */
  async acceptInvite(token: string, userId: string): Promise<AcceptInviteResponse> {
    const invite = await this.invitesRepo.findByToken(token);
    if (!invite) throw new NotFoundError('Invite not found.');

    if (invite.usedAt !== null) {
      throw new ConflictError('INVITE_ALREADY_USED', 'This invite has already been used.');
    }
    if (invite.expiresAt <= new Date()) {
      throw new GoneError('INVITE_EXPIRED', 'This invite has expired.');
    }

    const alreadyMember = await this.membersRepo.isMember(invite.teamId, userId);
    if (alreadyMember) {
      throw new ConflictError('ALREADY_A_MEMBER', 'You are already a member of this team.');
    }

    const { team } = await this.membersRepo.addMemberWithRoles(
      invite.teamId,
      userId,
      invite.roles,
      invite.id,
    );

    await this.auditRepo.record({
      teamId:     invite.teamId,
      actorId:    userId,
      event:      'member_invited',
      entityType: 'team',
      entityId:   invite.teamId,
      metadata:   { newUserId: userId, roles: invite.roles },
    });

    return { team, roles: invite.roles };
  }

  /**
   * Lists all pending (unused, unexpired) invites for a team.
   * The token field is never returned to callers of this method.
   *
   * @param teamId - Team whose pending invites to retrieve.
   */
  async listPendingInvites(teamId: string): Promise<InviteListItem[]> {
    return this.invitesRepo.findPendingByTeam(teamId);
  }

  /**
   * Revokes a pending invite by hard-deleting its row.
   *
   * @param teamId   - Team context (prevents cross-team revocation).
   * @param inviteId - UUID of the invite to delete.
   * @throws {NotFoundError} If the invite is not found or already used.
   */
  async revokeInvite(teamId: string, inviteId: string): Promise<void> {
    const deleted = await this.invitesRepo.deleteById(inviteId, teamId);
    if (!deleted) throw new NotFoundError('Invite not found or already used.');
  }
}

import { prisma } from '../../shared/db/client';
import type { MemberListItem } from './members.types';

/**
 * All database access for team_members and team_member_roles.
 */
export class MembersRepository {
  /**
   * Checks whether a user is already a member of a team.
   *
   * @param teamId - Team to check.
   * @param userId - User to check.
   * @returns true if the user is a member.
   */
  async isMember(teamId: string, userId: string): Promise<boolean> {
    const row = await prisma.teamMember.findFirst({
      where: { teamId, userId },
      select: { id: true },
    });
    return row !== null;
  }

  /**
   * Returns the roles held by a user in a team.
   *
   * @param teamId - Team context.
   * @param userId - User whose roles to fetch.
   * @returns Array of role strings (empty if not a member).
   */
  async getRolesForUser(teamId: string, userId: string): Promise<string[]> {
    const rows = await prisma.teamMemberRole.findMany({
      where: { teamMember: { teamId, userId } },
      select: { role: true },
    });
    return rows.map((r) => r.role);
  }

  /**
   * Atomically inserts a team_member row and assigns roles.
   * Also marks the invite as used (usedAt = now()). Run inside a transaction
   * to guarantee atomicity; this method does NOT open its own transaction —
   * the caller (service) is responsible if atomicity is needed.
   *
   * @param teamId   - Team to join.
   * @param userId   - User joining.
   * @param roles    - Role strings to assign.
   * @param inviteId - Invite to mark used (optional; undefined for seed data).
   * @returns The team row (id + name).
   */
  async addMemberWithRoles(
    teamId: string,
    userId: string,
    roles: string[],
    inviteId?: string,
  ): Promise<{ team: { id: string; name: string } }> {
    return prisma.$transaction(async (tx) => {
      const member = await tx.teamMember.create({
        data: { teamId, userId },
      });

      for (const role of roles) {
        await tx.teamMemberRole.create({ data: { teamMemberId: member.id, role } });
      }

      if (inviteId) {
        await tx.teamInvite.update({
          where: { id: inviteId },
          data: { usedAt: new Date() },
        });
      }

      const team = await tx.team.findUniqueOrThrow({
        where: { id: teamId },
        select: { id: true, name: true },
      });

      return { team };
    });
  }

  /**
   * Lists all members of a team with their roles and join date.
   *
   * @param teamId - Team to list.
   * @returns Array of MemberListItem.
   */
  async listMembersWithRoles(teamId: string): Promise<MemberListItem[]> {
    const memberRows = await prisma.teamMember.findMany({
      where: { teamId },
      include: {
        user: { select: { email: true } },
        roles: { select: { role: true } },
      },
    });

    return memberRows.map((m) => ({
      userId:   m.userId,
      email:    m.user.email,
      roles:    m.roles.map((r) => r.role),
      joinedAt: m.createdAt.toISOString(),
    }));
  }

  /**
   * Replaces all existing roles for a member with a new set.
   * Runs inside a transaction to avoid partial updates.
   *
   * @param teamId - Team context.
   * @param userId - User whose roles to replace.
   * @param roles  - New role set (must be non-empty).
   * @returns The userId and new roles, or undefined if the user is not a member.
   */
  async updateRoles(
    teamId: string,
    userId: string,
    roles: string[],
  ): Promise<{ userId: string; roles: string[] } | undefined> {
    return prisma.$transaction(async (tx) => {
      const member = await tx.teamMember.findFirst({
        where: { teamId, userId },
      });
      if (!member) return undefined;

      await tx.teamMemberRole.deleteMany({ where: { teamMemberId: member.id } });
      for (const role of roles) {
        await tx.teamMemberRole.create({ data: { teamMemberId: member.id, role } });
      }
      return { userId, roles };
    });
  }

  /**
   * Removes a member from a team by deleting the team_members row.
   * Cascades to delete team_member_roles automatically via FK.
   *
   * @param teamId - Team context.
   * @param userId - User to remove.
   * @returns true if removed, false if not found.
   */
  async removeMember(teamId: string, userId: string): Promise<boolean> {
    const member = await prisma.teamMember.findFirst({ where: { teamId, userId } });
    if (!member) return false;
    await prisma.teamMember.delete({ where: { id: member.id } });
    return true;
  }
}

// apps/api/src/teams/members/members.service.ts
import { MembersRepository } from './members.repository';
import { AuditRepository } from '../../shared/audit/audit.repository';
import type { UpdateRolesDto, MemberListItem } from './members.types';
import { NotFoundError, ForbiddenError } from '../../shared/errors/http-errors';

/**
 * Business logic for team member management.
 * Enforces guards (cannot remove owner, cannot self-lockout) before delegating to repository.
 */
export class MembersService {
  constructor(
    private readonly membersRepo: MembersRepository,
    private readonly auditRepo: AuditRepository,
  ) {}

  /**
   * Lists all members of a team with their roles.
   *
   * @param teamId - Team to list.
   */
  async listMembers(teamId: string): Promise<MemberListItem[]> {
    return this.membersRepo.listMembersWithRoles(teamId);
  }

  /**
   * Replaces a member's role set. Enforces:
   * - Cannot modify the owner's roles
   * - Actor cannot strip their own admin/owner access (SELF_LOCKOUT)
   *
   * @param teamId  - Team context.
   * @param actorId - The authenticated user performing the update.
   * @param userId  - The user whose roles are being updated.
   * @param dto     - New roles to assign.
   * @returns Updated userId and new roles.
   * @throws {ForbiddenError} On owner protection or self-lockout.
   * @throws {NotFoundError}  If target user is not a member.
   */
  async updateMemberRoles(
    teamId: string,
    actorId: string,
    userId: string,
    dto: UpdateRolesDto,
  ): Promise<{ userId: string; roles: string[] }> {
    const targetRoles = await this.membersRepo.getRolesForUser(teamId, userId);
    if (targetRoles.includes('owner')) {
      throw new ForbiddenError('CANNOT_MODIFY_OWNER', 'Cannot modify the team owner\'s roles.');
    }

    // Self-lockout: actor removes their own admin (or owner, though owner is blocked above)
    if (actorId === userId) {
      const actorRoles = await this.membersRepo.getRolesForUser(teamId, actorId);
      const wouldLoseElevated =
        (actorRoles.includes('admin') && !dto.roles.includes('admin')) ||
        (actorRoles.includes('owner') && !dto.roles.includes('owner'));
      if (wouldLoseElevated) {
        throw new ForbiddenError('SELF_LOCKOUT', 'Cannot remove your own admin access.');
      }
    }

    const oldRoles = await this.membersRepo.getRolesForUser(teamId, userId);
    const result = await this.membersRepo.updateRoles(teamId, userId, dto.roles);
    if (!result) throw new NotFoundError('Member not found.');

    await this.auditRepo.record({
      teamId,
      actorId,
      event:      'member_role_updated',
      entityType: 'team',
      entityId:   teamId,
      metadata:   { targetUserId: userId, oldRoles, newRoles: dto.roles },
    });

    return result;
  }

  /**
   * Removes a member from the team. Enforces:
   * - Cannot remove the owner
   * - Actor cannot remove themselves
   *
   * @param teamId  - Team context.
   * @param actorId - Authenticated user performing the removal.
   * @param userId  - User to remove.
   * @throws {ForbiddenError} On owner protection or self-removal.
   * @throws {NotFoundError}  If target is not a member.
   */
  async removeMember(teamId: string, actorId: string, userId: string): Promise<void> {
    if (actorId === userId) {
      throw new ForbiddenError('CANNOT_REMOVE_SELF', 'You cannot remove yourself from the team.');
    }

    const targetRoles = await this.membersRepo.getRolesForUser(teamId, userId);
    if (targetRoles.includes('owner')) {
      throw new ForbiddenError('CANNOT_REMOVE_OWNER', 'Cannot remove the team owner.');
    }

    const removed = await this.membersRepo.removeMember(teamId, userId);
    if (!removed) throw new NotFoundError('Member not found.');

    await this.auditRepo.record({
      teamId,
      actorId,
      event:      'member_removed',
      entityType: 'team',
      entityId:   teamId,
      metadata:   { removedUserId: userId },
    });
  }
}
